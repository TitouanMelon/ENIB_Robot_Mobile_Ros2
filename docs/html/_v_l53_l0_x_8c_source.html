311// Set the return signal rate limit check value in units of MCPS (mega counts
312// per second). "This represents the amplitude of the signal reflected from the
313// target and detected by the device"; setting this limit presumably determines
314// the minimum measurement necessary for the sensor to report a valid reading.
315// Setting a lower limit increases the potential range of the sensor but also
316// seems to increase the likelihood of getting an inaccurate reading because of
317// unwanted reflections from objects other than the intended target.
318// Defaults to 0.25 MCPS as initialized by the ST API and this library.
319uint8_t setSignalRateLimit(float limit_Mcps)
320{
321 if (limit_Mcps < 0 || limit_Mcps > 511.99) { return false; }
322
323 // Q9.7 fixed point format (9 integer bits, 7 fractional bits)
324 writeReg16Bit(FINAL_RANGE_CONFIG_MIN_COUNT_RATE_RTN_LIMIT, limit_Mcps * (1 << 7));
325 return 0;
326}
327
328// Get the return signal rate limit check value in MCPS
329float getSignalRateLimit(void)
330{
331 return (float)readReg16Bit(FINAL_RANGE_CONFIG_MIN_COUNT_RATE_RTN_LIMIT) / (1 << 7);
332}

338// Performs a single-shot range measurement and returns the reading in
339// millimeters
340// based on VL53L0X_PerformSingleRangingMeasurement()
341// extraStats provides additional info for this measurment. Set to 0 if not needed.
342uint16_t readRangeSingleMillimeters( /*statInfo_t *extraStats */) {
343 writeReg(0x80, 0x01);
344 writeReg(0xFF, 0x01);
345 writeReg(0x00, 0x00);
346 writeReg(0x91, g_stopVariable);
347 writeReg(0x00, 0x01);
348 writeReg(0xFF, 0x00);
349 writeReg(0x80, 0x00);
350 writeReg(SYSRANGE_START, 0x01);
351
352 uint16_t temp;
353
354 if (ACTIVE_WHILE)
355 {
356 while (readReg(SYSRANGE_START) & 0x01){};
357 while ((readReg(RESULT_INTERRUPT_STATUS) & 0x07) == 0){};
358 }
359 // assumptions: Linearity Corrective Gain is 1000 (default);
360 // fractional ranging is not enabled
361 temp = readReg16Bit(RESULT_RANGE_STATUS + 10);
362 temp+=0;
363
364 return temp;
365}
