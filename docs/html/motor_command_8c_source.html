7static TIM_HandleTypeDef TimHandle;
8static TIM_OC_InitTypeDef sConfigOC;
9
10//=================================================================
11// PWM INIT
12// TIMER 3 (PWM) : CH1 et CH2
13// ENABLE : Sortie Logique (GPIO) PA7
14//=================================================================
15
16void motorCommand_Init(void)
17{
18 unsigned int uwPrescalerValue = 0;
19
20 /* Compute the prescaler value to have TIM4 counter clock equal to 10MHz */
21 uwPrescalerValue = (unsigned int) ((SystemCoreClock / 10000000) - 1);
22 TimHandle.Instance = TIM3;
23 TimHandle.Init.Period = 200 - 1; // 100MHz/200=50kHz
24 TimHandle.Init.Prescaler = uwPrescalerValue;
25 TimHandle.Init.ClockDivision = 0;
26 TimHandle.Init.CounterMode = TIM_COUNTERMODE_UP;
27
28 HAL_TIM_Base_Init(&TimHandle);
29
30 sConfigOC.OCMode = TIM_OCMODE_PWM1;
31 sConfigOC.Pulse = 0x5;// Specifies the pulse value to be loaded into the Capture Compare Register. This parameter can be a number between Min_Data = 0x0000 and Max_Data = 0xFFFF */
32
33 sConfigOC.OCPolarity = TIM_OCPOLARITY_HIGH;
34 sConfigOC.OCFastMode = TIM_OCFAST_DISABLE;
35
36 HAL_TIM_PWM_ConfigChannel(&TimHandle, &sConfigOC, TIM_CHANNEL_1);
37 HAL_TIM_PWM_ConfigChannel(&TimHandle, &sConfigOC, TIM_CHANNEL_2);
38
39 // CHANGEMENT DU RAPPORT CYCLIQUE
40 __HAL_TIM_SetCompare(&TimHandle, TIM_CHANNEL_1, 100); // 100 : moteurs au repos
41 __HAL_TIM_SetCompare(&TimHandle, TIM_CHANNEL_2, 100);
42
43 HAL_TIM_PWM_Start(&TimHandle, TIM_CHANNEL_1); // MOTOR RIGHT
44 HAL_TIM_PWM_Start(&TimHandle, TIM_CHANNEL_2); // MOTOR LEFT
45
46 // ENABLE MOTEUR (SI INVERSEUR)
47 //HAL_GPIO_WritePin(GPIOA, GPIO_PIN_7, 0);
48 HAL_GPIO_WritePin(GPIOB, GPIO_PIN_3, 0);
49}
