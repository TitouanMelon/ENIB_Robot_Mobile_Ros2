385void task_Motor_Left(void *pvParameters)
386{
387 int16_t consigne = 0; //Store the desirate speed
388
389 float ui = 0.0; //Integral term of the correcteur
390 float up = 0.0; //Proportionnal term of the correcteur
391 int err = 0; //Error term of the correcteur
392 int speed = 0; //Actual speed of motor
393
394 for (;;)
395 {
396 xQueueReceive(q_mot_L, &consigne, portMAX_DELAY); //receive wanted speed
397
398 speed = quadEncoder_GetSpeedL(); //Get actual speed
399 //Calculate term of correcteur
400 err=consigne-speed;
401 up=LKp*(float)err;
402 ui=ui+LKp*LKi*(float)err;
403
404 motorLeft_SetDuty(100+(int)(up+ui)); //Set duty cycle of the motor
405
406 xSemaphoreGive(xSem_Supervision); //Give semaphore to liberate the decision task
407 vTaskDelay(SAMPLING_PERIOD_ms);
408 }

614 speedLeft = VITESSE_KART+(8*(speed-50));
615 speedRight = -(VITESSE_KART+(8*(speed-50)));
616 break;
617 case GAUCHE:
618 speedLeft = -(VITESSE_KART+(8*(speed-50)));
619 speedRight = VITESSE_KART+(8*(speed-50));
620 break;
621 case AVANT_GAUCHE:
622 speedLeft = (VITESSE_KART/2)+(8*(speed-50));
623 speedRight = VITESSE_KART+(8*(speed-50));
624 break;
625 case AVANT_DROITE:
626 speedLeft = VITESSE_KART+(8*(speed-50));
627 speedRight = (VITESSE_KART/2)+(8*(speed-50));
628 break;
629 case RECULE_GAUCHE:
630 speedLeft = -(VITESSE_KART+(8*(speed-50)));
631 speedRight = -((VITESSE_KART/2)+(8*(speed-50)));
632 break;
633 case RECULE_DROITE:
634 speedLeft = -((VITESSE_KART/2)+(8*(speed-50)));
635 speedRight = -(VITESSE_KART+(8*(speed-50)));
636 break;
637 default:
638 speedLeft = 0;
639 speedRight = 0;
640 break;
641 }
642 }
643 else if (mode == MODE_OBS) //Mode obstacle
644 {
645 //Get sensors informations
646 captDistIR_Get(table);
647 #if VL53
648 if (uxQueueMessagesWaiting(qhVl53))
649 xQueueReceive(qhVl53, &vl53, portMAX_DELAY);
650 else
651 vl53 = 0;
652
653 if (vl53 == 1) //if an obstacle is detected on the back we stop
654 {
655 if (dir != 'S')
656 printf("Detection d'un obstacle à l'arrièrre");
657 speedLeft = 0;
658 speedRight = 0;
659 dir = 'S';
660 obs = 1;
661 }
662 else
663 #endif //VL53
664 if (table[0] > SEUIL_DIST_SENSOR || table[1] > SEUIL_DIST_SENSOR) //We have an obstacle in front of the robot
665 {
666 if (obs > 10) //If we detect more than 10 different obstacle we turn on the left until they are no more obstacle
667 {
668 speedLeft = VITESSE_OBS/2;
669 speedRight = -VITESSE_OBS/2;
670 dir = 'G';
671 }
672 else
673 {
674 speedLeft = 0;
675 speedRight = 0;
676
677 if (table[0] > table[1] && table[0] > SEUIL_DIST_SENSOR) //We have an obstacle on our right
678 {
679 dir = 'G';
680 speedLeft = -VITESSE_OBS/2;
681 speedRight = VITESSE_OBS/2;
682 if (obs%2 == 0)
683 obs++;
684 }
685 else if (table[0] < table[1] && table[1] > SEUIL_DIST_SENSOR) //We have an obstacle on left right
686 {
687 dir = 'D';
688 speedLeft = VITESSE_OBS/2;
689 speedRight = -VITESSE_OBS/2;
690 if (obs%2 == 1)
691 obs++;
692 }
693 }
694 }
695 else //No obstacle
696 {
697 speedLeft = VITESSE_OBS;
698 speedRight = VITESSE_OBS;
699 dir = 'F';
700 obs = 0;
701 }
702 }
703 else if (mode == MODE_CAM) //Mode camera
704 {
705 dir = 'N';
706 obs = 0;
707
708 if(x < 0 || y < 0) //No object
709 {
710 speedLeft = 0;
711 speedRight = 0;
712 }
713 else //Try to keep the object on the center
714 {
715 speedLeft = VITESSE_CAM - ((CAMERA_X_MAX/2 - x))/3; // (int) (((float) ((x-CAMERA_X_MAX/2)/CAMERA_X_MAX))*500);
716 speedRight = VITESSE_CAM + ((CAMERA_X_MAX/2 - x))/3; // (int) (((float) (x/CAMERA_X_MAX))*500);
717 }
718 }
719
720 #if DEBUG_MOTOR
721 printf("Motor L : %d || R : %d\r\n", speedLeft, speedRight);
722 #endif
723
724 xQueueSend( q_mot_L, ( void * ) &speedLeft, portMAX_DELAY ); //Send motor left speed
725 xSemaphoreTake( xSem_Supervision, portMAX_DELAY );
726
727 xQueueSend( q_mot_R, ( void * ) &speedRight, portMAX_DELAY ); //Send motor right speed
728 xSemaphoreTake( xSem_Supervision, portMAX_DELAY );
729
730 #if MICROROS
731 MsgToPub.dir = dir;
732 MsgToPub.mode = mode;
733 MsgToPub.speed = speed;
734 if (!uxQueueMessagesWaiting(qhMR_pub)) //If no data in queue
735 xQueueSend(qhMR_pub, ( void * ) &MsgToPub, portMAX_DELAY); //Send data
736 #endif //MICROROS
737
738 #if LCD
739 if (!uxQueueMessagesWaiting(qhLCD)) //If no data in queue
740 {
741 pxMessage.data=mode;
742 pxMessage.command=dir;
743 xQueueSend( qhLCD, ( void * ) &pxMessage, portMAX_DELAY); //Send data
744 }
745 #endif //LCD
746
747 vTaskDelay(SAMPLING_PERIOD_ms);
748 }
749#endif //SYNCHRO_EX
750}
