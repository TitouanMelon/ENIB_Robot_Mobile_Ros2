17void captDistIR_Init(void)
18{
19 adcHandle.Instance = ADC1;
20
21 adcHandle.Init.ClockPrescaler = ADC_CLOCKPRESCALER_PCLK_DIV2;
22 adcHandle.Init.DataAlign = ADC_DATAALIGN_RIGHT;
23 adcHandle.Init.Resolution = ADC_RESOLUTION12b;
24 // Don't do continuous conversions - do them on demand
25 adcHandle.Init.ContinuousConvMode = DISABLE; // Continuous mode disabled to have only 1 conversion at each conversion trig
26 // Disable the scan conversion so we do one at a time */
27 adcHandle.Init.ScanConvMode = DISABLE;
28 //Say how many channels would be used by the sequencer
29 adcHandle.Init.NbrOfConversion = 2;
30 adcHandle.Init.DiscontinuousConvMode = DISABLE; // Parameter discarded because sequencer is disabled
31 adcHandle.Init.NbrOfDiscConversion = 2;
32 adcHandle.Init.ExternalTrigConvEdge = ADC_EXTERNALTRIGCONVEDGE_NONE ;
33 //Start conversion by software, not an external trigger
34 adcHandle.Init.ExternalTrigConv = 0;
35 adcHandle.Init.DMAContinuousRequests = DISABLE;
36 adcHandle.Init.EOCSelection = DISABLE;
37
38 HAL_ADC_Init(&adcHandle);
39}
